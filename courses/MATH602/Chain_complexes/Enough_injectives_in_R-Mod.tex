\begin{definition}
Suppose $\mathscr A,\mathscr B$ are abelian category, $F:\mathscr A\to \mathscr B$ is an \textbf{additive funtor}\index{Additive funtor} if $F(A\oplus B)$ is naturally isomorphic to $F(A)\oplus F(B)$, and $F$ is \textbf{additive}, i.e. $F(f+g)=F(f)+F(g)$, $f,g\in Hom(A,B)$, here $f+g$ is given by the composition $A\xrightarrow{\begin{pmatrix}
f \\ g
\end{pmatrix}} B\oplus B\xrightarrow{\nabla}Y$, here $\nabla$ is the \textbf{codiagonal}
\end{definition}

\begin{example}
$\bigwedge^2:\mathrm{Vect}_F\to\mathrm{Vect}_F$ is exact but not additive
\end{example}
